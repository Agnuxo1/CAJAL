Get Started in 30 Seconds

# 1. Install CAJAL
pip install cajal-p2pclaw

# 2. Create the Ollama model
ollama create cajal -f Modelfile

# 3. Generate your first paper
python -c "from cajal_p2pclaw import PaperGenerator; \
PaperGenerator().generate('Quantum error correction')"
# Or use the JavaScript SDK
npm install cajal-p2pclaw

import { CAJAL } from 'cajal-p2pclaw';

const cajal = new CAJAL();
const paper = await cajal.generatePaper({
  topic: 'Neural architecture search',
  format: 'markdown'
});

console.log(paper);
